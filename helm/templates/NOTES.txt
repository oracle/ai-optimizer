{{- /* Show the auto-wired endpoint URLs only when they're actually used.
       The deployment's per-signal precedence is: an explicit per-signal
       endpoint wins over the generic; the generic wins over the SigNoz
       auto-default. Also gate on the active exporter for each signal —
       tracesExporter=console/none means the app does not OTLP-export
       traces, and logsEnabled=false (the default) means the app never
       activates log export. Surfacing the auto-URL in those cases reads
       as "wired" but nothing actually flows to it. */ -}}
{{- $otelEnabled := .Values.server.otel.enabled -}}
{{- $endpointEmpty := eq (.Values.server.otel.endpoint | default "" | trim) "" -}}
{{- $tracesEpEmpty := eq (.Values.server.otel.tracesEndpoint | default "" | trim) "" -}}
{{- $logsEpEmpty := eq (.Values.server.otel.logsEndpoint | default "" | trim) "" -}}
{{- /* Mirror the validator's per-token parse: empty/whitespace defaults
       to "otlp"; otherwise OTLP is active only when "otlp" appears as a
       token. Trim each token so "otlp, console" still resolves cleanly.
       Traces stays case-sensitive — the app drops non-lowercase traces
       tokens via set-intersection. Logs is case-insensitive because the
       app and validator both lowercase log tokens before checking. */ -}}
{{- $tracesRaw := .Values.server.otel.tracesExporter | default "" | trim -}}
{{- $tracesUsesOtlp := eq $tracesRaw "" -}}
{{- range (splitList "," $tracesRaw) -}}{{- if eq (trim .) "otlp" -}}{{- $tracesUsesOtlp = true -}}{{- end -}}{{- end -}}
{{- $logsRaw := .Values.server.otel.logsExporter | default "" | trim -}}
{{- $logsUsesOtlp := eq $logsRaw "" -}}
{{- range (splitList "," $logsRaw) -}}{{- if eq (lower (trim .)) "otlp" -}}{{- $logsUsesOtlp = true -}}{{- end -}}{{- end -}}
{{- $autoTracesEp := "" -}}
{{- $autoLogsEp := "" -}}
{{- if and $otelEnabled $tracesUsesOtlp $endpointEmpty $tracesEpEmpty -}}
{{- $autoTracesEp = include "server.otel.defaultTracesEndpoint" . -}}
{{- end -}}
{{- if and $otelEnabled .Values.server.otel.logsEnabled $logsUsesOtlp $endpointEmpty $logsEpEmpty -}}
{{- $autoLogsEp = include "server.otel.defaultLogsEndpoint" . -}}
{{- end -}}
{{- if or (ne $autoTracesEp "") (ne $autoLogsEp "") }}

In-cluster OTLP collector endpoints wired into the server:
{{- if ne $autoTracesEp "" }}
  traces: {{ $autoTracesEp }}
{{- end }}
{{- if ne $autoLogsEp "" }}
  logs:   {{ $autoLogsEp }}
{{- end }}

UPGRADE & UNINSTALL
================================================================================

To upgrade this release (substitute <chart-source> with the path, repo
reference, or OCI URL you used to install — bare "{{ .Chart.Name }}" is
interpreted by Helm as a repo name and will fail):
  helm upgrade {{ .Release.Name }} <chart-source> -n {{ .Release.Namespace }} --reuse-values

  # Examples:
  #   helm upgrade {{ .Release.Name }} ./{{ .Chart.Name }} -n {{ .Release.Namespace }} --reuse-values    # local path
  #   helm upgrade {{ .Release.Name }} myrepo/{{ .Chart.Name }} -n {{ .Release.Namespace }} --reuse-values  # repo

To uninstall this release:
  helm uninstall {{ .Release.Name }} -n {{ .Release.Namespace }}
